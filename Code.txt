// =============== Pin Map ===============
// GPS NEO-6M  (UART1)
#define GPS_RX_PIN     32   // ESP32 RX  <- GPS TX
#define GPS_TX_PIN     33   // ESP32 TX  -> GPS RX
// Flysky iA6B IBUS (UART2 RX only)
#define IBUS_RX_PIN    16
// ACS712-20A through 10k/20k divider

// Left motor (L298N)

#define IN1            27

// Right motor (L298N)

// =============== PWM (Core 3.x.x) ===============
#define PWM_FREQ      1000
#define PWM_RES       8

// =============== Tunables ===============
const float    ACS_SENSITIVITY = 0.0666f;   // V/A after voltage divider
const float    LEAK_THRESHOLD  = 0.5f;      // amps
const uint32_t TELEGRAM_INTERVAL_MS = 5000;
const uint32_t IBUS_TIMEOUT_MS      = 500;

// =============== Globals ===============

SemaphoreHandle_t stateMutex;

// shared state — always read/write under stateMutex
float   g_zeroVoltage = 1.65f;
float   g_lastCurrent = 0.0f;
bool    g_leakActive  = false;
double  g_lat = 0.0, g_lng = 0.0;
bool    g_gpsFix = false;

// =============== Helpers ===============
static String urlEncode(const String& s) {
  String out; out.reserve(s.length() * 3);
  const char* hex = "0123456789ABCDEF";
  for (size_t i = 0; i < s.length(); ++i) {
    char c = s[i];
    if (isalnum((unsigned char)c) || c == '-' || c == '_' || c == '.' || c == '~') {
      out += c;
    } else {
      out += '%';
      out += hex[(c >> 4) & 0xF];
      out += hex[c & 0xF];
    }
  }
  return out;
}

void calibrateACS() {
  Serial.println(F("[ACS] Calibrating zero — disconnect load..."));
  long sum = 0;
  for (int i = 0; i < 500; i++) {
    sum += analogRead(ACS_PIN);
    delay(2);
  }
  float avg = sum / 500.0f;
  float zv  = (avg / 4095.0f) * 3.3f;
  if (xSemaphoreTake(stateMutex, portMAX_DELAY) == pdTRUE) {
    g_zeroVoltage = zv;
    xSemaphoreGive(stateMutex);
  }
  Serial.printf("[ACS] Zero voltage = %.3f V\n", zv);
}

// อัปเดตฟังก์ชันขับมอเตอร์ ให้สั่งงานตรงไปที่ pwmPin
void driveMotor(int pwmPin, int inA, int inB, int speed) {
  speed = constrain(speed, -255, 255);
  if (speed >= 0) { 
    digitalWrite(inA, HIGH); digitalWrite(inB, LOW);  
  } else { 
    digitalWrite(inA, LOW);  digitalWrite(inB, HIGH); speed = -speed; 
  }
  ledcWrite(pwmPin, speed); // Core 3.x.x: สั่งความเร็วไปที่ Pin โดยตรง
}

void stopMotors() {
  digitalWrite(IN1, LOW); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, LOW);
  ledcWrite(ENA, 0); // Core 3.x.x: สั่งไปที่ ENA
  ledcWrite(ENB, 0); // Core 3.x.x: สั่งไปที่ ENB
}

bool sendTelegram(const String& text) {
  if (WiFi.status() != WL_CONNECTED) return false;
  WiFiClientSecure client;
  client.setInsecure();          
  HTTPClient http;
  String url = "https://api.telegram.org/bot";
  url += BOT_TOKEN;
  url += "/sendMessage?chat_id=";
  url += CHAT_ID;
  url += "&text=";
  url += urlEncode(text);

  http.setConnectTimeout(4000);
  http.setTimeout(4000);
  if (!http.begin(client, url)) return false;
  int code = http.GET();
  http.end();
  Serial.printf("[TG] HTTP %d\n", code);
  return code == 200;
}

// =============== Core 0: Network task ===============
void networkTask(void* arg) {
  WiFi.mode(WIFI_STA);
  WiFi.setAutoReconnect(true);
  WiFi.begin(WIFI_SSID, WIFI_PASS);

  uint32_t t0 = millis();
  while (WiFi.status() != WL_CONNECTED && millis() - t0 < 20000) {
    vTaskDelay(pdMS_TO_TICKS(500));
  }
  if (WiFi.status() == WL_CONNECTED) {
    Serial.print(F("[WiFi] Connected: "));
    Serial.println(WiFi.localIP());
    sendTelegram("Boat online — leak monitor active.");
  } else {
    Serial.println(F("[WiFi] Initial connect failed; auto-reconnect on."));
  }

  uint32_t lastTgMs = 0;

  for (;;) {
    while (GPSSerial.available()) gps.encode(GPSSerial.read());

    if (gps.location.isUpdated() && gps.location.isValid()) {
      double lat = gps.location.lat();
      double lng = gps.location.lng();
      if (xSemaphoreTake(stateMutex, pdMS_TO_TICKS(10)) == pdTRUE) {
        g_lat = lat; g_lng = lng; g_gpsFix = true;
        xSemaphoreGive(stateMutex);
      }
    }

    bool   leak = false, fix = false;
    float  cur  = 0.0f;
    double lat  = 0.0, lng = 0.0;
    if (xSemaphoreTake(stateMutex, pdMS_TO_TICKS(10)) == pdTRUE) {
      leak = g_leakActive;
      cur  = g_lastCurrent;
      lat  = g_lat; lng = g_lng; fix = g_gpsFix;
      xSemaphoreGive(stateMutex);
    }

    if (leak && (millis() - lastTgMs >= TELEGRAM_INTERVAL_MS)) {
      String msg = "⚠️ Electrical leak detected!\n";
      msg += "Current: " + String(cur, 2) + " A\n";
      if (fix) {
        msg += "Location: https://maps.google.com/?q=";
        msg += String(lat, 6); msg += ","; msg += String(lng, 6);
      } else {
        msg += "GPS: no fix";
      }
      sendTelegram(msg);
      lastTgMs = millis();
    }

    vTaskDelay(pdMS_TO_TICKS(50));   
  }
}

// =============== Setup ===============
void setup() {
  Serial.begin(115200);
  delay(200);

  analogReadResolution(12);
  analogSetPinAttenuation(ACS_PIN, ADC_11db);

  // Motor pins
  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);
  
  // อัปเดตเป็น API ของ ESP32 Core 3.x.x
  ledcAttach(ENA, PWM_FREQ, PWM_RES);
  ledcAttach(ENB, PWM_FREQ, PWM_RES);
  
  stopMotors();

  // Servos
  servoBase.setPeriodHertz(50);
  servoPitch.setPeriodHertz(50);
  servoGripper.setPeriodHertz(50);
  servoBase.attach(SERVO_BASE_PIN,    500, 2400);
  servoPitch.attach(SERVO_PITCH_PIN,  500, 2400);
  servoGripper.attach(SERVO_GRIPPER_PIN, 500, 2400);
  servoBase.write(90); servoPitch.write(90); servoGripper.write(90);

  // GPS UART1
  GPSSerial.begin(9600, SERIAL_8N1, GPS_RX_PIN, GPS_TX_PIN);

  // IBUS on UART2 (RX-only on pin 16)
  Serial2.begin(115200, SERIAL_8N1, IBUS_RX_PIN, -1);
  ibus.begin(Serial2, IBUSBM_NOTIMER);    

  stateMutex = xSemaphoreCreateMutex();

  calibrateACS();

  xTaskCreatePinnedToCore(
      networkTask, "net", 8192, nullptr, 1, nullptr, 0);

  Serial.println(F("[BOOT] Setup complete."));
}

// =============== Core 1: Realtime control ===============
void loop() {
  ibus.loop();

  static uint32_t lastFrameCnt = 0;
  static uint32_t lastFrameMs  = 0;
  if (ibus.cnt_rec != lastFrameCnt) {
    lastFrameCnt = ibus.cnt_rec;
    lastFrameMs  = millis();
  }
  bool linkLost = (millis() - lastFrameMs > IBUS_TIMEOUT_MS);

  if (linkLost) {
    stopMotors();
  } else {
    int ch1 = ibus.readChannel(0);  // steering
    int ch2 = ibus.readChannel(1);  // throttle
    int ch3 = ibus.readChannel(2);  // pitch
    int ch5 = ibus.readChannel(4);  // base
    int ch6 = ibus.readChannel(5);  // gripper

    ch1 = constrain(ch1, 1000, 2000);
    ch2 = constrain(ch2, 1000, 2000);
    ch3 = constrain(ch3, 1000, 2000);
    ch5 = constrain(ch5, 1000, 2000);
    ch6 = constrain(ch6, 1000, 2000);

    int throttle = map(ch2, 1000, 2000, -255, 255);
    int steering = map(ch1, 1000, 2000, -255, 255);

    if (abs(throttle) < 15) throttle = 0;   
    if (abs(steering) < 15) steering = 0;

    int leftSpd  = constrain(throttle + steering, -255, 255);
    int rightSpd = constrain(throttle - steering, -255, 255);

    // ส่งชื่อขา Pin (ENA, ENB) ไปตรงๆ ในฟังก์ชัน driveMotor (Core 3.x.x)
    driveMotor(ENA, IN1, IN2, leftSpd);
    driveMotor(ENB, IN3, IN4, rightSpd);

    servoBase.write(   map(ch5, 1000, 2000, 0, 180));
    servoPitch.write(  map(ch3, 1000, 2000, 0, 180));
    servoGripper.write(map(ch6, 1000, 2000, 0, 180));
  }

  uint32_t accum = 0;
  const int N = 8;
  for (int i = 0; i < N; i++) accum += analogRead(ACS_PIN);
  float v = ((accum / (float)N) / 4095.0f) * 3.3f;

  float zero;
  if (xSemaphoreTake(stateMutex, pdMS_TO_TICKS(5)) == pdTRUE) {
    zero = g_zeroVoltage;
    xSemaphoreGive(stateMutex);
  } else {
    zero = 1.65f;
  }
  float current = (v - zero) / ACS_SENSITIVITY;

  if (xSemaphoreTake(stateMutex, pdMS_TO_TICKS(5)) == pdTRUE) {
    g_lastCurrent = current;
    g_leakActive  = (fabsf(current) > LEAK_THRESHOLD);
    xSemaphoreGive(stateMutex);
  }

  delay(20);   
}
