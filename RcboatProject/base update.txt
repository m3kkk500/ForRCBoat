#define GPS_RX_PIN        32  // ขารับสัญญาณจากโมดูล GPS NEO-7M
#define ZMPT_PIN          34  // ขารับสัญญาณอนาล็อกคลื่นไซน์จากเซ็นเซอร์ ZMPT101B (ADC1_CH6)

// บอร์ดควบคุมมอเตอร์ขับเคลื่อนหลัก L298N 
#define ENA_PIN           26  // สัญญาณ PWM คุมความเร็วฝั่งซ้าย
#define IN1_PIN           27  // ทิศทางมอเตอร์ซ้าย 1
#define IN2_PIN           14  // ทิศทางมอเตอร์ซ้าย 2
#define ENB_PIN           23  // สัญญาณ PWM คุมความเร็วฝั่งขวา
#define IN3_PIN           18  // ทิศทางมอเตอร์ขวา 1 (แก้ไขแล้ว ป้องกันการชนกับพินเซอร์โว)
#define IN4_PIN           19  // ทิศทางมอเตอร์ขวา 2 (แก้ไขแล้ว ปลดล็อกพิน 33)

// เซอร์โวมอเตอร์คุมแขนกล MG996R (พัดกระแสผ่านราง UBEC 6V แยกตัว)
#define SERVO_BASE_PIN    13  // พินคุมเซอร์โวฐานเลี้ยว
#define SERVO_PITCH_PIN   21  // พินคุมเซอร์โวยกแขนขึ้น-ลง
#define SERVO_GRIP_PIN    25  // พินคุมเซอร์โวคีมจับสิ่งปฏิกูล

// ==================== [ข้อมูลและค่าตัวแปรอ้างอิงหลัก] ====================
const char* WIFI_SSID     = "YOUR_WIFI_SSID";
const char* WIFI_PASSWORD = "YOUR_WIFI_PASSWORD";
const char* TELEGRAM_BOT_TOKEN = "YOUR_BOT_TOKEN";
const char* TELEGRAM_CHAT_ID   = "YOUR_CHAT_ID";

const float ZMPT_CALIBRATION_COEFF = 0.095;   // สัมประสิทธิ์สอบเทียบหาค่า RMS จริง
const float LEAK_THRESHOLD_VOLTS  = 30.0;    // จุดจำกัดความปลอดภัยในการแจ้งเตือน (30V AC)

// โครงสร้างแชร์ข้อมูลข้ามสองแกนประมวลผล (Thread-Safety Shared Memory)
struct NavigationData {
  double currentLat = 0.0;
  double currentLng = 0.0;
  double homeLat = 0.0;
  double homeLng = 0.0;
  bool isHomeSet = false;
  
  float leakVoltageRMS = 0.0;
  bool isLeakAlert = false;
  bool rthActive = false;
};

volatile NavigationData navData;
SemaphoreHandle_t xDataMutex = NULL; // ตัวล็อก Mutex ป้องกันสิทธิ์การเขียนอ่านพร้อมกัน

TinyGPSPlus gps;
IBusBM ibus;
Servo servoBase;
Servo servoPitch;
Servo servoGripper;

const int pwmFreq = 1000;
const int pwmResolution = 8;
const int channelLeft = 0;
const int channelRight = 1;

TaskHandle_t TaskGPS_Handle = NULL;
TaskHandle_t TaskTelegram_Handle = NULL;

void driveMotor(int ch, int in1, int in2, int speed);
float calculateTrueRMS();
String urlEncode(String str);

// ==================== ====================
void setup() {
  Serial.begin(115200);
  Serial1.begin(9600, SERIAL_8N1, GPS_RX_PIN, -1); // รับค่าจากจีพีเอสผ่านซีเรียล 1
  Serial2.begin(115200, SERIAL_8N1, 16, -1);      // รับสัญญาณแชนแนลคอนโทรล iBUS (RX เท่านั้น)

  xDataMutex = xSemaphoreCreateMutex();
  if (xDataMutex == NULL) {
    Serial.println("System Error: Semaphore initialization failed!");
    while(1);
  }

  // ตั้งค่าเอาต์พุตวงจรขับมอเตอร์ L298N
  pinMode(IN1_PIN, OUTPUT);
  pinMode(IN2_PIN, OUTPUT);
  pinMode(IN3_PIN, OUTPUT);
  pinMode(IN4_PIN, OUTPUT);
  
  ledcAttach(ENA_PIN, pwmFreq, pwmResolution);
  ledcAttach(ENB_PIN, pwmFreq, pwmResolution);

  // การจัดตั้งคลาสตระกูลควบคุมเซอร์โว
  ESP32PWM::allocateTimer(0);
  ESP32PWM::allocateTimer(1);
  servoBase.setPeriodHertz(50);
  servoPitch.setPeriodHertz(50);
  servoGripper.setPeriodHertz(50);
  
  servoBase.attach(SERVO_BASE_PIN, 500, 2400);
  servoPitch.attach(SERVO_PITCH_PIN, 500, 2400);
  servoGripper.attach(SERVO_GRIP_PIN, 500, 2400);

  analogReadResolution(12);

  // เริ่มต้นรับข้อมูลจากเครื่องรับ FS-iA6B เฉพาะขาเข้า
  ibus.begin(Serial2, IBUSBM_NOTIMER);

  // ยึดแยกพอร์ตการทำงานหนักไปคุมที่ Core 0
  xTaskCreatePinnedToCore(TaskGPS, "TaskGPS", 4096, NULL, 3, &TaskGPS_Handle, 0);
  xTaskCreatePinnedToCore(TaskTelegram, "TaskTelegram", 8192, NULL, 1, &TaskTelegram_Handle, 0);

  Serial.println("System initiated. Telemetry and Battery Monitor bypassed.");
}

// ==================== ====================
void loop() {
  ibus.loop();
  bool isConnected = (ibus.readChannel(0)!= 0);

  double latNow = 0.0, lngNow = 0.0, hLat = 0.0, hLng = 0.0;
  bool hSet = false, currentRthRequest = false;

  if (xSemaphoreTake(xDataMutex, pdMS_TO_TICKS(10)) == pdTRUE) {
    latNow = navData.currentLat;
    lngNow = navData.currentLng;
    hLat = navData.homeLat;
    hLng = navData.homeLng;
    hSet = navData.isHomeSet;
    currentRthRequest = navData.rthActive;
    xSemaphoreGive(xDataMutex);
  }

  // คำนวณรากกำลังสองเฉลี่ยยอดคลื่นไฟสลับ AC ของโพรบวัด ZMPT101B
  float currentVoltage = calculateTrueRMS();
  bool leakAlertStatus = (currentVoltage >= LEAK_THRESHOLD_VOLTS);

  // ส่งผลการประเมินวิเคราะห์แรงดันไฟฟั่วไหลเข้าหน่วยความจำเพื่อแชร์ให้งาน Telegram
  if (xSemaphoreTake(xDataMutex, pdMS_TO_TICKS(10)) == pdTRUE) {
    navData.leakVoltageRMS = currentVoltage;
    navData.isLeakAlert = leakAlertStatus;
    xSemaphoreGive(xDataMutex);
  }

  // ระบบประมวลผล RTH หลุดระยะควบคุม / หรือสั่ง RTH ด้วยมือ
  if (!isConnected || currentRthRequest) {
    if (hSet) {
      double distanceToHome = TinyGPSPlus::distanceBetween(latNow, lngNow, hLat, hLng);
      if (distanceToHome > 5.0) {
        int rthThrottle = 150; 
        int rthSteering = 0; // วิ่งตรงชั่วคราวเพื่อตั้งลำพิกัดทิศทาง (COG)
        
        driveMotor(channelLeft, IN1_PIN, IN2_PIN, rthThrottle + rthSteering);
        driveMotor(channelRight, IN3_PIN, IN4_PIN, rthThrottle - rthSteering);
      } else {
        driveMotor(channelLeft, IN1_PIN, IN2_PIN, 0);
        driveMotor(channelRight, IN3_PIN, IN4_PIN, 0);
      }
    } else {
      driveMotor(channelLeft, IN1_PIN, IN2_PIN, 0);
      driveMotor(channelRight, IN3_PIN, IN4_PIN, 0);
    }
  } else {
    // ระบบบังคับพวงมาลัยและแขนกลแบบไร้สายปกติ
    int ch1 = constrain(ibus.readChannel(0), 1000, 2000); // ข้อมูลสเตียริ่งซ้ายขวา
    int ch2 = constrain(ibus.readChannel(1), 1000, 2000); // แรงบิดความเร็วเดินหน้า
    int ch3 = constrain(ibus.readChannel(2), 1000, 2000); // คุม Servo Pitch
    int ch5 = constrain(ibus.readChannel(4), 1000, 2000); // คุม Servo Base
    int ch6 = constrain(ibus.readChannel(5), 1000, 2000); // คุม Servo Gripper
    int ch4 = constrain(ibus.readChannel(3), 1000, 2000); // สวิตช์สั่งนำเรือกลับ RTH

    int throttle = map(ch2, 1000, 2000, -255, 255);
    int steering = map(ch1, 1000, 2000, -180, 180);

    if (abs(throttle) < 15) throttle = 0;
    if (abs(steering) < 15) steering = 0;

    int leftSpeed = constrain(throttle + steering, -255, 255);
    int rightSpeed = constrain(throttle - steering, -255, 255);

    driveMotor(channelLeft, IN1_PIN, IN2_PIN, leftSpeed);
    driveMotor(channelRight, IN3_PIN, IN4_PIN, rightSpeed);

    // ควบคุมกลไกข้อต่อพวงแขนคีบขยะผิวน้ำ
    servoBase.write(map(ch5, 1000, 2000, 0, 180));
    servoPitch.write(map(ch3, 1000, 2000, 0, 180));
    servoGripper.write(map(ch6, 1000, 2000, 0, 180));

    bool rthToggle = (ch4 > 1700);
    if (xSemaphoreTake(xDataMutex, pdMS_TO_TICKS(10)) == pdTRUE) {
      navData.rthActive = rthToggle;
      xSemaphoreGive(xDataMutex);
    }
  }

  vTaskDelay(pdMS_TO_TICKS(1));
}

// ==================== [ฟังก์ชันสนับสนุนระบบการขับเครื่องยนต์] ====================
void driveMotor(int ch, int in1, int in2, int speed) {
  if (speed > 0) {
    digitalWrite(in1, HIGH);
    digitalWrite(in2, LOW);
    ledcWrite(ch, speed);
  } else if (speed < 0) {
    digitalWrite(in1, LOW);
    digitalWrite(in2, HIGH);
    ledcWrite(ch, abs(speed));
  } else {
    digitalWrite(in1, LOW);
    digitalWrite(in2, LOW);
    ledcWrite(ch, 0);
  }
}

// คำนวณ True RMS คลื่น AC ตลอดความคาบคลื่นความถี่ 50Hz (20ms sampling window)
float calculateTrueRMS() {
  unsigned long startTime = millis();
  long sumOfSquares = 0;
  int sampleCount = 0;
  static float dynamicDCOffset = 2048.0; 

  while (millis() - startTime < 20) { 
    int rawVal = analogRead(ZMPT_PIN);
    dynamicDCOffset = (0.999 * dynamicDCOffset) + (0.001 * rawVal);
    float acComponent = rawVal - dynamicDCOffset;
    sumOfSquares += (acComponent * acComponent);
    sampleCount++;
    delayMicroseconds(150);
  }

  if (sampleCount == 0) return 0.0;

  float meanSquare = (float)sumOfSquares / sampleCount;
  float rmsADC = sqrt(meanSquare);
  float finalVoltage = rmsADC * ZMPT_CALIBRATION_COEFF;
  
  if (finalVoltage < 2.0) finalVoltage = 0.0; 
  return finalVoltage;
}

// ==================== ====================
void TaskGPS(void *pvParameters) {
  (void) pvParameters;
  for (;;) {
    while (Serial1.available() > 0) {
      char c = Serial1.read();
      if (gps.encode(c)) {
        if (gps.location.isValid()) {
          double currentLat = gps.location.lat();
          double currentLng = gps.location.lng();
          
          if (xSemaphoreTake(xDataMutex, pdMS_TO_TICKS(15)) == pdTRUE) {
            navData.currentLat = currentLat;
            navData.currentLng = currentLng;
            
            if (!navData.isHomeSet && gps.satellites.value() >= 5) {
              navData.homeLat = currentLat;
              navData.homeLng = currentLng;
              navData.isHomeSet = true;
              Serial.println("Safe Haven Locked via GPS.");
            }
            xSemaphoreGive(xDataMutex);
          }
        }
      }
    }
    vTaskDelay(pdMS_TO_TICKS(10));
  }
}

void TaskTelegram(void *pvParameters) {
  (void) pvParameters;
  WiFiClientSecure client;
  client.setInsecure(); // บล็อกระบบการตรวจสอบใบรับรองความปลอดภัยเพื่อความเร็วสัญญาณข้ามเซิร์ฟเวอร์

  unsigned long lastAlertTime = 0;
  const unsigned long alertCooldown = 15000; // ตั้งค่าหน่วงยิงสแปมซ้ำซ้อนรอบละ 15 วินาที

  for (;;) {
    if (WiFi.status()!= WL_CONNECTED) {
      WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
      int retryCount = 0;
      while (WiFi.status()!= WL_CONNECTED && retryCount < 10) {
        vTaskDelay(pdMS_TO_TICKS(1000));
        retryCount++;
      }
    }

    bool activeAlert = false;
    float currentVolt = 0.0;
    double latSend = 0.0;
    double lngSend = 0.0;

    if (xSemaphoreTake(xDataMutex, pdMS_TO_TICKS(20)) == pdTRUE) {
      activeAlert = navData.isLeakAlert;
      currentVolt = navData.leakVoltageRMS;
      latSend = navData.currentLat;
      lngSend = navData.currentLng;
      xSemaphoreGive(xDataMutex);
    }

    if (activeAlert && (millis() - lastAlertTime > alertCooldown) && WiFi.status() == WL_CONNECTED) {
      HTTPClient http;
      String url = "https://api.telegram.org/bot" + String(TELEGRAM_BOT_TOKEN) + "/sendMessage";
      
      String message = "🚨 [ตรวจพบกระแสไฟฟ้ารั่วไหลในแหล่งน้ำ] 🚨\n";
      message += "⚡ พิกัดความแรงดันรั่วไหล: " + String(currentVolt, 1) + " V AC\n";
      message += "📍 พิกัดเรือลาดตระเวน: " + String(latSend, 6) + ", " + String(lngSend, 6) + "\n";
      message += "🔗 ตรวจสอบตำแหน่ง Google Maps: https://maps.google.com/?q=" + String(latSend, 6) + "," + String(lngSend, 6) + "\n";
      message += "⚠️ คำแนะนำ: ร้องเรียนตัดไฟทันที! ระงับกิจกรรมทางน้ำบริเวณนี้";

      http.begin(client, url);
      http.addHeader("Content-Type", "application/x-www-form-urlencoded");
      
      String httpRequestData = "chat_id=" + String(TELEGRAM_CHAT_ID) + "&text=" + urlEncode(message);
      int httpResponseCode = http.POST(httpRequestData);
      
      if (httpResponseCode > 0) {
        Serial.println("Telegram Notification Dispatched Successfully.");
        lastAlertTime = millis();
      }
      http.end();
    }
    vTaskDelay(pdMS_TO_TICKS(2000));
  }
}

String urlEncode(String str) {
  String encodedString = "";
  char c;
  char code0;
  char code1;
  for (int i = 0; i < str.length(); i++) {
    c = str.charAt(i);
    if (c == ' ') {
      encodedString += '+';
    } else if (isalnum(c)) {
      encodedString += c;
    } else {
      code1 = (c & 0xf) + '0';
      if ((c & 0xf) > 9) {
        code1 = (c & 0xf) - 10 + 'A';
      }
      c = (c >> 4) & 0xf;
      code0 = c + '0';
      if (c > 9) {
        code0 = c - 10 + 'A';
      }
      encodedString += '%';
      encodedString += code0;
      encodedString += code1;
    }
  }
  return encodedString;
}
